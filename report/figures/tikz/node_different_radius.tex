% Definition of radii
\def\smallradius{(0,0) circle (0.75cm)}
\def\largeradius{(1.5,0) circle (1.5cm)}

\colorlet{circle edge}{black!50}

\tikzset{
  filled/.style={fill=circle area, draw=circle edge, thick},
  outline/.style={draw=circle edge, thick},
  dash/.style={very thick, dashed}
}

\tikzstyle{vertex}=[circle,fill=black!25,minimum size=10pt,inner sep=0pt]

\setlength{\parskip}{5mm}
\begin{tikzpicture}
  \draw[outline] \smallradius node {};
  \draw[dash] \largeradius {};

  \draw \smallradius node {};
  \draw \largeradius node {};

  \node[vertex] (v) at (0,0) {v};
  \node[vertex] (u) at (1.5, 0) {u};
\end{tikzpicture}
